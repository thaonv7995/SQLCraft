SELECT

u.user_id,

u.email,

SUM(t.amount) AS total_revenue,

COUNT(t.id) AS transaction_count

FROM users u

JOIN transactions t ON u.user_id = t.user_id

WHERE t.created_at >= '2023-01-01'

GROUP BY 1, 2

HAVING total_revenue > 1000

ORDER BY total_revenue DESC;
